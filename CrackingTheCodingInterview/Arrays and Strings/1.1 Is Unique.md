Pg. 90 (101): Implement and algirhtm to determine if a string has all unique characters.

#### Advice:
- Clarify what the character set is - shows an attention to detail and can help give clues

## Sorting:

```c++
class Solution {
public:
	bool uniqueCharacters(string s) {
		sort(s.begin(), s.end());
		int len = s.length();
		for (int i = 0; i < len - 1; ++i) {
			if (s[i] == s[i + 1]) return false;
		}
		return true;
	}
}
```

- O(nlogn) Time
- O(1) Space (Dependant on sorting algorithm)

## HashSet:

```c++
class Solution {
public:
	bool uniqueCharacters(string s) {
		unordered_set<char> seen {};
		for (auto ch : s) {
			if (seen.count(s)) return false;
			seen.insert(s);
		}
		return true;
	}
}

// Without using unordered_set
class Solution {
public:
	bool uniqueCharacters(string s) {
		int seen[256] {};
		for (auto ch : s) {
			if (seen[ch]) return false;
			seen[ch]++;
		}
		return true;
	}
}
```

- O(n) Time, or O(c) where c is the size of the character set
- O(1) Space
